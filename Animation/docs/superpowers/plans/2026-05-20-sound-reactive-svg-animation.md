```javascript
const audio = {
  ctx: null,
  analyser: null,
  dataArray: null,
  smoothVal: 0,
  smoothing: 0.3,
  rms: function() {
    this.analyser.getByteTimeDomainData(this.dataArray);
    let sum = 0;
    const len = this.dataArray.length;
    for (let i = 0; i < len; i++) {
      const val = this.dataArray[i] / 128 - 1;
      sum += val * val;
    }
    const raw = Math.sqrt(sum / len);
    this.smoothVal += (raw - this.smoothVal) * this.smoothing;
    return this.smoothVal;
  }
};

async function initAudio() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    audio.ctx = new (window.AudioContext || window.webkitAudioContext)();
    const source = audio.ctx.createMediaStreamSource(stream);
    audio.analyser = audio.ctx.createAnalyser();
    audio.analyser.fftSize = 256;
    source.connect(audio.analyser);
    audio.dataArray = new Uint8Array(audio.analyser.frequencyBinCount);
  } catch (e) {
    console.warn('Microphone access denied:', e);
  }
}
```

```javascript
const els = {
  air: document.querySelector('#Air'),
  towel: document.querySelector('#Towel'),
  napkin: document.querySelector('#Napkin'),
  hair: document.querySelector('#Hair'),
};

const airPaths = els.air ? Array.from(els.air.querySelectorAll('path')) : [];

let time = 0;
let windTimer = 0;        // counts down from 4s when sound triggers
let windIntensity = 0;    // smoothed 0-1, decays after timer
const THRESHOLD = 0.12;   // sound level to trigger wind
const WIND_DURATION = 4;  // seconds

function animate() {
  requestAnimationFrame(animate);
  const dt = 0.016; // ~60fps per frame
  time += dt;

  const level = audio.ctx ? audio.rms() : 0;

  // Trigger wind if sound exceeds threshold
  if (level > THRESHOLD) {
    windTimer = WIND_DURATION;
  }

  // Decay wind timer
  if (windTimer > 0) {
    windTimer -= dt;
    // Intensity rises quickly, then decays as timer runs out
    windIntensity = Math.min(windIntensity + dt * 3, Math.min(windTimer / WIND_DURATION * 2, 1));
  } else {
    // Smoothly ease back to rest
    windIntensity *= 0.97;
  }

  const i = Math.max(0, windIntensity);

  // Air: hidden by default, visible + vibrate on wind
  if (els.air) {
    const airOpacity = Math.min(i * 1.5, 0.8);
    els.air.style.opacity = airOpacity;
    airPaths.forEach((path, idx) => {
      const phase = idx * 1.7;
      const rot = i * 8 * Math.sin(time * 3 + phase);
      const dy = i * 6 * Math.sin(time * 4 + phase + 1);
      path.style.transform = `rotate(${rot}deg) translateY(${dy}px)`;
      path.style.transformOrigin = 'center center';
    });
  }

  // Towel: wave up-left
  if (els.towel) {
    const tRot = -i * 6 * Math.sin(time * 2.5);
    const tX = -i * 4 * Math.abs(Math.sin(time * 2.3));
    const tY = -i * 5 * Math.abs(Math.sin(time * 2.7));
    els.towel.style.transform = `translate(${tX}px,${tY}px) rotate(${tRot}deg)`;
    els.towel.style.transformOrigin = 'center center';
  }

  // Napkin: wave up-left with offset
  if (els.napkin) {
    const nRot = -i * 4 * Math.sin(time * 2.2 + 1.2);
    const nX = -i * 3 * Math.abs(Math.sin(time * 2.0 + 0.5));
    const nY = -i * 4 * Math.abs(Math.sin(time * 2.4 + 0.8));
    els.napkin.style.transform = `translate(${nX}px,${nY}px) rotate(${nRot}deg)`;
    els.napkin.style.transformOrigin = 'center center';
  }

  // Hair: slow gentle wave
  if (els.hair) {
    const hRot = i * 3 * Math.sin(time * 1.2);
    els.hair.style.transform = `rotate(${hRot}deg)`;
    els.hair.style.transformOrigin = 'center center';
  }
}

animate();
```
